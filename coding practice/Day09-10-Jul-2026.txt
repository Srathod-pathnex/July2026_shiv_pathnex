# installing docker compose using Ansible
---
- name: Install Docker Compose
  hosts: all
  become: yes

  tasks: 
    - name: Install dependencies
      yum: 
        name: "python3-pip"
        state: present
    - name: Install docker-compose
      pip:
        name: Docker compose
