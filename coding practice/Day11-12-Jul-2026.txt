# Install helm on  server

---
- name: Install Helm
  hosts: all
  become: yes
  tasks:
    - name: Download Helm
      get_url:
         url: https://get.helm.sh/helm-v3.7.0-linux-amd64.tar.gz
         dest: /tem/heml.tar.gz
    - name: Extract Helm
      unarchive:
         src: /tem/helm.tar.gz
         dest: /user/local/bin/
         remote_src: yes
    - name: Set Helm binary permission
      file:
         path: /user/local/bin/helm/
         mode: "0755"


#Dockerfile

FROM nginx
EXPOSE 80
RUN docker run -d -p 8080:80 nginx
